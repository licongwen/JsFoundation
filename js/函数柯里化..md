
## 函数柯里化

```js
function curry(fn){
    if(fn.length<=1) return fn;
    const generator = (...args)=>{
        if(args.length === fn.length){
            return fn(...args);
        }else{
            return (...args2)=>{
                return generator(...args,...args2)
            }
        }
    }
    return generator;
}

//使用案例
function add(a,b,c,d){
    return a+b+c+d;
}
let curryAdd = curry(add);

let sum = curryAdd(1)(2)(3)(4)

console.log(sum)//10
```
