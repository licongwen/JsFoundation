## 乱序一个数组

### 代码如下

```js
    let arr = [1,2,3,4,5,6,7,8,9,0];   
    function getArr(arr){
        let arr1 =[];
        for(let i=0;i<10;i++){
            let num = Math.floor(Math.random()*10 - i)
            arr1.push(arr.splice(num,1)[0])
        }
        return arr1;
    }
    let sortArr = getArr(arr);
```
