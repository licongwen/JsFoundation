### 快速排序
```js
function quickSort(arr){
    if(arr.length<=1){
        return arr;
    }
    let len = arr.length;
    let mid = arr.splice(Math.floor(len/2),1);//基准数，不一定事中间的一个
    let leftArr=[],rightArr=[];
    arr.map(item=>{
        if(item>=mid[0]){
            rightArr.push(item)
        }else{
            leftArr.push(item)
        }
    })
    return quickSort(leftArr).concat(mid,quickSort(rightArr));
}
```
